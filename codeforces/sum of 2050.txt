const int sz = 1e5;
bool isPrime[sz + 1];
 
// Function for Sieve of Eratosthenes
bool iszero(int x)
{
	while(x!=0)
	{
		int l=x%10;
		if(l==0)
		{
			return false;
		}
		x=x/10;
	}
	return true;
}
void sieve()
{
	memset(isPrime, true, sizeof(isPrime));
 
	isPrime[0] = isPrime[1] = false;
 
	for (int i = 2; i * i <= sz; i++) {
		if (isPrime[i]) {
			for (int j = i * i; j < sz; j += i) {
				isPrime[j] = false;
			}
		}
	}
}

		if (isPrime[i]) {

{
   if(n==0)
   return false;

 bool isPerfectSquare(long double x)
{
  
    if (x >= 0) {
 
        long long sr = sqrt(x);
         
      
        return (sr * sr == x);
    }
    
    return false;
}
